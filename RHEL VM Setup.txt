- IP ADDRESSING


ip address add 10.0.0.101/16 dev ens192
ip route add 10.0.0.101 via 10.0.0.1 dev ens192



vi /etc/sysconfig/network-scripts/ifcfg-ens192


DEVICE=ens192
HWADDR=00:0c:29:84:6d:8c
NM_CONTROLLED=no
ONBOOT=yes
IPADDR=10.0.0.101
BOOTPROTO=none
NETMASK=255.255.0.0
DNS=10.0.0.1
TYPE=Ethernet
IPV6INIT=no
USERCTL=no
